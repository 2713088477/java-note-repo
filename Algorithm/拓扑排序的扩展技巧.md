## 题目1:最大食物链计数

代码:

```java
import java.io.*;
import java.util.Arrays;

//测试链接:https://www.luogu.com.cn/problem/P4017
public class Code01_Luogu4017 {
    public static long mod = 80112002L;
    public static int MAX_NODE = (int)5E3+1;
    public static int MAX_EDGE = (int)5e5+1;
    //链式前向星
    public static int[] head = new int[MAX_NODE];
    public static int[] next = new int[MAX_EDGE];
    public static int[] to = new int[MAX_EDGE];
    public static int edgeId = 1;
    //标签信息
    public static long[] lines = new long[MAX_NODE];
    //入度表
    public static int[] indegree = new int[MAX_NODE];
    //队列
    public static int[] deque = new int[MAX_NODE];
    public static int l=0,r=0;
    public static int n,m;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StreamTokenizer in = new StreamTokenizer(br);
        PrintWriter out = new PrintWriter(new OutputStreamWriter(System.out));
        while(in.nextToken() != StreamTokenizer.TT_EOF){
            n = (int)in.nval;
            in.nextToken();
            m = (int)in.nval;
            for(int i=0,fromNode,toNode;i<m;i++){
                in.nextToken();
                fromNode = (int)in.nval;
                in.nextToken();
                toNode = (int)in.nval;
                addEdge(fromNode, toNode);
            }
            out.println(count());
        }
        clear();
        out.flush();
        out.close();
        br.close();
    }
    public static void addEdge(int fromNode,int toNode){
        indegree[toNode]++;
        next[edgeId] = head[fromNode];
        to[edgeId] = toNode;
        head[fromNode] = edgeId++;
        
    }
    public static long count(){
        long ans = 0;
        for(int i=1;i<=n;i++){
            if(indegree[i]==0){
                lines[i] = 1;
                if(head[i]==0){
                    ans  = (ans + lines[i])%mod;
                }
                deque[r++] = i;
            }
        }
        int fir=0;
        while(l<r){
            fir = deque[l++];
            for(int e=head[fir];e>0;e=next[e]){
                lines[to[e]] = (lines[to[e]]%mod + lines[fir]%mod)%mod;
                if(--indegree[to[e]] == 0){
                    if(head[to[e]]==0){
                        ans  = (ans + lines[to[e]])%mod;
                    }
                    deque[r++] = to[e];
                }
                
            }
        }
        return ans%mod;
    }
    public static void clear(){
        Arrays.fill(head,1,n+1,0);
        Arrays.fill(next,1,m+1,0);
        Arrays.fill(to,1,m+1,0);
        edgeId = 1;
        Arrays.fill(lines,1,n+1,0);
        Arrays.fill(indegree,1,n+1,0);
        l=0;r=0;

    }
}

```

给每个节点加上lines[]标签，依次统计到最后

## 题目2：喧闹与富有

代码:

```java
import java.util.ArrayList;

//测试链接:https://leetcode.cn/problems/loud-and-rich/
public class Code02_Solution851 {
    public int[] loudAndRich(int[][] richer, int[] quiet) {
        int n = quiet.length;
        ArrayList<ArrayList<Integer>> edge = new ArrayList<>();
        for(int i=0;i<n;i++){
            edge.add(new ArrayList<>());
        }
        int[] indegree = new int[n];
        for(int i=0,fromNode,toNode;i<richer.length;i++){
            fromNode = richer[i][0];
            toNode = richer[i][1];
            edge.get(fromNode).add(toNode);
            indegree[toNode]++;
        }
        int[] deque = new int[n];
        int l=0,r=0;
        int[] ans = new int[n];
        for(int i=0;i<n;i++){
            ans[i] = i;
        }
        for(int i=0;i<indegree.length;i++){
            if(indegree[i]==0){
                deque[r++] = i;
            }
        }
        int pop;
        while(l<r){
            pop = deque[l++];
            for(int e:edge.get(pop)){
                ans[e] = quiet[ans[pop]]<quiet[ans[e]] ? ans[pop]:ans[e];
                if(--indegree[e]==0){
                    deque[r++] = e;
                }
            }
        }
        return ans;
        
    }
}

```

这道题也是，每一个边在处理的时候传递信息给下一个节点。

这道题是，传递当前quiet值最小的person下去。

## 题目3:并行课程III

代码:

```java
class Solution {
    public int minimumTime(int n, int[][] relations, int[] time) {
        ArrayList<ArrayList<Integer>> edge = new ArrayList<>(n+1);
        for(int i=0;i<=n;i++){
            edge.add(new ArrayList<>());
        }
        int[] indegree = new int[n+1];
        for(int i=0,f,t;i<relations.length;i++){
            f = relations[i][0];
            t = relations[i][1];
            edge.get(f).add(t);
            indegree[t]++;
        }
        int[] curTime = new int[n+1];
        int[] deque = new int[n];
        int l=0,r=0;
        for(int i=1;i<indegree.length;i++){
            if(indegree[i]==0){
                deque[r++] = i;
            }
        }
        int ans = 0;
        int pop;
        while(l<r){
            pop = deque[l++];
            curTime[pop] += time[pop-1];
            ans = Math.max(ans,curTime[pop]);
            for(int t:edge.get(pop)){
                curTime[t] = Math.max(curTime[t],curTime[pop]);
                if(--indegree[t] ==0 ){
                    deque[r++] = t;
                }
            } 
        }
        return ans;
    }
}
```
