# 拓扑排序的扩展技巧

## 精炼总结

重要技巧: <font color="red">利用拓扑排序的过程，上游节点逐渐推送消息给下游节点的技巧</font>

## 题目1:最大食物链计数

代码:

```java
import java.io.*;
import java.util.Arrays;

//测试链接:https://www.luogu.com.cn/problem/P4017
public class Code01_Luogu4017 {
    public static long mod = 80112002L;
    public static int MAX_NODE = (int)5E3+1;
    public static int MAX_EDGE = (int)5e5+1;
    //链式前向星
    public static int[] head = new int[MAX_NODE];
    public static int[] next = new int[MAX_EDGE];
    public static int[] to = new int[MAX_EDGE];
    public static int edgeId = 1;
    //标签信息
    public static long[] lines = new long[MAX_NODE];
    //入度表
    public static int[] indegree = new int[MAX_NODE];
    //队列
    public static int[] deque = new int[MAX_NODE];
    public static int l=0,r=0;
    public static int n,m;
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StreamTokenizer in = new StreamTokenizer(br);
        PrintWriter out = new PrintWriter(new OutputStreamWriter(System.out));
        while(in.nextToken() != StreamTokenizer.TT_EOF){
            n = (int)in.nval;
            in.nextToken();
            m = (int)in.nval;
            for(int i=0,fromNode,toNode;i<m;i++){
                in.nextToken();
                fromNode = (int)in.nval;
                in.nextToken();
                toNode = (int)in.nval;
                addEdge(fromNode, toNode);
            }
            out.println(count());
        }
        clear();
        out.flush();
        out.close();
        br.close();
    }
    public static void addEdge(int fromNode,int toNode){
        indegree[toNode]++;
        next[edgeId] = head[fromNode];
        to[edgeId] = toNode;
        head[fromNode] = edgeId++;
        
    }
    public static long count(){
        long ans = 0;
        for(int i=1;i<=n;i++){
            if(indegree[i]==0){
                lines[i] = 1;
                if(head[i]==0){
                    ans  = (ans + lines[i])%mod;
                }
                deque[r++] = i;
            }
        }
        int fir=0;
        while(l<r){
            fir = deque[l++];
            for(int e=head[fir];e>0;e=next[e]){
                lines[to[e]] = (lines[to[e]]%mod + lines[fir]%mod)%mod;
                if(--indegree[to[e]] == 0){
                    if(head[to[e]]==0){
                        ans  = (ans + lines[to[e]])%mod;
                    }
                    deque[r++] = to[e];
                }
                
            }
        }
        return ans%mod;
    }
    public static void clear(){
        Arrays.fill(head,1,n+1,0);
        Arrays.fill(next,1,m+1,0);
        Arrays.fill(to,1,m+1,0);
        edgeId = 1;
        Arrays.fill(lines,1,n+1,0);
        Arrays.fill(indegree,1,n+1,0);
        l=0;r=0;

    }
}

```

给每个节点加上lines[]标签，依次统计到最后
